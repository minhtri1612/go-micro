Thank you for installing {{ .Chart.Name }}.

Your PostgreSQL database has been deployed with these details:

   Host: {{ include "order-db.fullname" . }}
   Port: {{ .Values.service.port }}
   Database: {{ .Values.auth.database }}
   Username: {{ .Values.auth.username }}

To connect to your database, run:

   # Get the PostgreSQL password:
   export POSTGRES_PASSWORD=$(kubectl get secret --namespace {{ .Release.Namespace }} {{ include "order-db.fullname" . }} -o jsonpath="{.data.POSTGRES_PASSWORD}" | base64 --decode)

   # Run a PostgreSQL pod and connect using psql:
   kubectl run postgresql-client --rm --tty -i --restart='Never' \
   --namespace {{ .Release.Namespace }} \
   --image postgres:14-alpine \
   --env="PGPASSWORD=$POSTGRES_PASSWORD" \
   --command -- psql --host {{ include "order-db.fullname" . }} \
   --username {{ .Values.auth.username }} \
   --dbname {{ .Values.auth.database }} \
   --port {{ .Values.service.port }}
